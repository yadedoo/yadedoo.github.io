# 4、数据库实例安装

# 4.1、配置监听

netca /silent /responseFile /home/oratest/app/oracle/product/19.3.0/assistants/netca/netca.rsp
1

# 4.2、创建数据库

# 4.2.3、备份响应文件

cd /home/oratest/app/product/19.3.0/assistants/dbca
cp dbca.rsp dbca.rsp.bak
1
2

# 4.2.4、编辑响应文件

gdbName=ORATEST                                       -- 数据库全局名
sid=ORATEST                                           -- 数据库SID
databaseConfigType=SI                                 -- 单实例数据库,SI\RAC\RACONENODE
templateName=General_Purpose.dbc                      -- 通用数据库模板,General_Purpose.dbc\Data_Warehouse.dbc  
sysPassword=password                                  -- sys密码
systemPassword=passowrd                               -- system密码
datafileDestination=/home/oratest/app/oradata         -- 数据库数据库文件路径
characterSet=AL32UTF8                                 -- 数据库字符集
createAsContainerDatabase=true                        -- 是否创建为容器数据库：true\false
